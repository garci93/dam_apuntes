public class Controlador {

    private Vista vista;
    private Servicio servicio;

    public Controlador() {
        vista = new Vista();
        servicio = new Servicio();
    }

    public void menu() {
        int opcion;
        do {
            opcion = vista.seleccionarOpcion();
            float parametroUsuario;

            switch(opcion) {

            case 1: parametroUsuario = vista.parametroUsuario();
                    ArrayList<Inmueble> inmueblesVenta = servicio.inmueblesVenta(parametroUsuario);
                    vista.mostrarVentasPrecioInferior(inmueblesVenta);
                break;

            default: 
                break;
            }
        }while(opcion != 4);
    }
}

    public void mostrarVentasPrecioInferior(ArrayList<Inmueble> inmueblesVenta) {

        }
        System.out.println(texto);
    }

    }

}

public class Servicio implements IAgencia{

    private ArrayList<Inmueble> inmuebles;

    public Servicio() {
        this.inmuebles = DAO.getInstance().getInmueble();

    }
        @Override
    public ArrayList<Inmueble> inmueblesVenta(Float precio) {
        ArrayList<Inmueble> inmueblesVenta = new ArrayList<Inmueble>();
        for(int i=0; i<inmuebles.size(); i++) {
            if(inmuebles.get(i) instanceof Vivienda) {
                Vivienda vivienda = (Vivienda) inmuebles.get(i);
                if(vivienda.getPrecio() < precio) {
                    inmueblesVenta.add(inmuebles.get(i));
                }
            }
            if(inmuebles.get(i) instanceof Solar) {
                Solar solar = (Solar) inmuebles.get(i);
                if(solar.calcularPrecio() < precio) {
                    inmueblesVenta.add(inmuebles.get(i));
                }
            }
        }
        return inmueblesVenta;
    }
